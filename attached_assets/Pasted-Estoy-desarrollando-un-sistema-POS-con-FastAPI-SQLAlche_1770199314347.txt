BACKEND (OBLIGATORIO)
1. Nueva entidad: PaymentSchedule
class PaymentSchedule(Base):
    __tablename__ = "payment_schedules"

    id = Column(Integer, primary_key=True)
    supplier_id = Column(Integer, ForeignKey("suppliers.id"))
    supplier_invoice_id = Column(Integer, ForeignKey("supplier_invoices.id"))

    invoice_due_date = Column(Date)
    invoice_number = Column(String)
    invoice_amount = Column(Numeric(12,2))

    payment_type = Column(String)  # TOTAL | ABONO
    paid_amount = Column(Numeric(12,2), default=0)
    pending_amount = Column(Numeric(12,2))

    status = Column(String)  # PAGADA | PENDIENTE | PARCIAL

    week_start = Column(Date)
    week_end = Column(Date)

    created_at = Column(DateTime, default=datetime.utcnow)

2. Lógica automática

pending_amount = invoice_amount - paid_amount

Si pending_amount == 0 → status = PAGADA

Si paid_amount > 0 → status = PARCIAL

Si paid_amount == 0 → status = PENDIENTE
